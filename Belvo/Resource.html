def list(params: nil)
  results = []
  @session.list(@endpoint, params: params) { |item| results.push item }
  results
end

def resume(session_id:, token:, link: nil)
  data = { session: session_id, token: token, link: link }
  @session.patch(@endpoint, data)
end
